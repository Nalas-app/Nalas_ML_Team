```json
{

  "order_details": {

    "menu_item_name": "string",
    "quantity": "integer (servings)",

  "event_details": {

  },
  
  "prediction_data": {
    "predicted_ingredient_cost": "float",
    "predicted_labor_cost": "float",
    "predicted_overhead_cost": "float",
    "predicted_total_cost": "float",
    "confidence_score": "float (0-1)",
    "model_version": "string",
    "prediction_method": "rule_based | ml_model",
    "prediction_id": "uuid (links to prediction log)"
  },
  
  "actual_data": {
    "actual_total_cost": "float (filled after order completion)",
    "actual_ingredients_used": "JSON (optional — detailed ingredient usage)",
    "completion_date": "ISO 8601 date",
    "was_cost_adjusted": "boolean",

  "feedback": {
```

## 3. Database Schema Recommendations

### 3.1 Core Tables for ML Data Collection

```sql
-- Table: prediction_logs
-- Purpose: Log every ML prediction for training data
CREATE TABLE prediction_logs (
    prediction_id       UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    menu_item_id        UUID NOT NULL REFERENCES menu_items(id),
    quantity            INTEGER NOT NULL CHECK (quantity > 0),
    event_date          DATE,
    guest_count         INTEGER CHECK (guest_count > 0),
    
    -- Prediction output
    ingredient_cost     DECIMAL(12,2) NOT NULL,
    labor_cost          DECIMAL(12,2) NOT NULL,
    overhead_cost       DECIMAL(12,2) NOT NULL,
    total_cost          DECIMAL(12,2) NOT NULL,
    confidence_score    DECIMAL(4,2),
    model_version       VARCHAR(20) NOT NULL,
    prediction_method   VARCHAR(20) NOT NULL CHECK (prediction_method IN ('rule_based', 'ml_model')),
    
    -- Metadata
    latency_ms          INTEGER,
    flags               JSONB DEFAULT '[]',
    created_at          TIMESTAMP DEFAULT NOW()
);

-- Table: order_costs
-- Purpose: Track actual costs for completed orders
CREATE TABLE order_costs (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    order_id            UUID NOT NULL REFERENCES orders(id),
    prediction_id       UUID REFERENCES prediction_logs(prediction_id),
    menu_item_id        UUID NOT NULL REFERENCES menu_items(id),
    
    -- Order details
    quantity            INTEGER NOT NULL,
    event_date          DATE,
    event_type          VARCHAR(50),
    guest_count         INTEGER,
    location            VARCHAR(100),
    
    -- Predicted costs (snapshot from prediction time)
    predicted_total     DECIMAL(12,2),
    
    -- Actual costs (filled after completion)
    actual_total        DECIMAL(12,2),
    actual_ingredient   DECIMAL(12,2),
    
    -- Comparison
    prediction_error    DECIMAL(12,2) GENERATED ALWAYS AS (actual_total - predicted_total) STORED,
    error_percentage    DECIMAL(6,2),
    
    -- Metadata
    completed_at        TIMESTAMP,
    was_adjusted        BOOLEAN DEFAULT FALSE,
    adjustment_reason   TEXT,
    created_at          TIMESTAMP DEFAULT NOW()
);

-- Table: ingredient_price_history
-- Purpose: Track ingredient price changes over time
CREATE TABLE ingredient_price_history (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    ingredient_id       UUID NOT NULL REFERENCES ingredients(id),
    old_price           DECIMAL(10,2),
    new_price           DECIMAL(10,2) NOT NULL,
    change_pct          DECIMAL(6,2),
    changed_at          TIMESTAMP DEFAULT NOW(),
    changed_by          VARCHAR(100)
);

-- Table: model_registry
-- Purpose: Track ML model versions
CREATE TABLE model_registry (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    version             VARCHAR(20) NOT NULL UNIQUE,
    algorithm           VARCHAR(50) NOT NULL,
    features            JSONB NOT NULL,
    metrics             JSONB NOT NULL,
    file_path           VARCHAR(500),
    is_active           BOOLEAN DEFAULT FALSE,
    trained_on_records  INTEGER,
    deployed_at         TIMESTAMP,
    created_at          TIMESTAMP DEFAULT NOW()
);

-- Indexes for common queries
CREATE INDEX idx_prediction_logs_item ON prediction_logs(menu_item_id);
CREATE INDEX idx_prediction_logs_date ON prediction_logs(created_at);
CREATE INDEX idx_order_costs_item ON order_costs(menu_item_id);
CREATE INDEX idx_order_costs_prediction ON order_costs(prediction_id);
CREATE INDEX idx_price_history_ingredient ON ingredient_price_history(ingredient_id);
```

### 3.2 View: Training Dataset

```sql
-- View: ml_training_data
-- Purpose: Ready-to-use training dataset for ML model
CREATE VIEW ml_training_data AS
SELECT 
    oc.menu_item_id,
    mi.name AS item_name,
    mi.category,
    oc.quantity,
    oc.event_date,
    EXTRACT(MONTH FROM oc.event_date) AS event_month,
    oc.event_type,
    oc.guest_count,
    oc.location,
    oc.predicted_total,
    oc.actual_total,
    oc.prediction_error,
    oc.error_percentage,
    pl.ingredient_cost AS predicted_ingredient_cost,
    pl.confidence_score,
    pl.model_version,
    pl.prediction_method
FROM order_costs oc
JOIN menu_items mi ON oc.menu_item_id = mi.id
LEFT JOIN prediction_logs pl ON oc.prediction_id = pl.prediction_id
WHERE oc.actual_total IS NOT NULL
ORDER BY oc.completed_at DESC;
```

## 4. Data Quality Monitoring Plan

### 4.1 Automated Quality Checks

| Check | Frequency | Query / Logic | Alert On |
|-------|-----------|--------------|----------|
| Missing actual costs | Daily | `SELECT COUNT(*) FROM order_costs WHERE completed_at IS NOT NULL AND actual_total IS NULL` | > 0 for > 3 days |
| Stale ingredient prices | Weekly | `SELECT * FROM ingredients WHERE last_updated < NOW() - INTERVAL '30 days'` | > 10% of ingredients |
| New items without recipes | Daily | `SELECT * FROM menu_items WHERE id NOT IN (SELECT DISTINCT menu_item_id FROM recipes)` | Any new item |
| Price spikes | On update | `ABS(new_price - old_price) / old_price > 0.50` | Any spike > 50% |
| Prediction coverage | Weekly | `SELECT menu_item_id, COUNT(*) FROM prediction_logs GROUP BY 1 HAVING COUNT(*) < 5` | Items with < 5 predictions |
| Data completeness | Weekly | `SELECT COUNT(*) FILTER (WHERE event_date IS NOT NULL) / COUNT(*)::float FROM order_costs` | < 90% completeness |
| Duplicate orders | Daily | Check for same item + qty + date combinations | Any duplicate |
